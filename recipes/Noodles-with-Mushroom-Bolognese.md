ingredients: |
  <h3>Ingredients:</h3>
  <ul>
  <li>1 packet <a href="/products/Changs-Long-Life-Noodles/" title="Chang's Long Life Noodles" alt="Chang's Long Life Noodles">Chang's Long Life Noodles</a></li>
  <li>2 tbs <a href="/products/Changs-Sesame-Oil/" title="Chang's Sesame Oil" alt="Chang's Sesame Oil">Chang's Sesame Oil</a></li>
  <li>1 large carrot, diced</li>
  <li>1 celery stick, diced</li>
  <li>4 cloves garlic, minced</li>
  <li>3 tbs <a href="/products/Changs-Soy-Sauce-Light/" title="Chang's Soy Sauce Light" alt="Chang's Soy Sauce Light">Chang's Soy Sauce Light</a></li>
  <li>400g can diced tomatoes</li>
  <li>400g brown mushrooms, diced</li>
  <li>Thai basil leaves</li>
  <li>Fresh red chilli</li>
  </ul>
method: |
  <h3>Method:</h3>
  <ol>
  <li>In a wok place sesame oil, carrot, celery and garlic. Saut&eacute; for 3 minutes.</li>
  <li>Add mushrooms and further saut&eacute; for 2 minutes.</li>
  <li>Add tomatoes, soy sauce, pepper and optional chilli slices. Simmer for 10 minutes over low heat.</li>
  <li>In the mean time boil noodles for 3 minutes and drain, reserving half a cup of noodles water.</li>
  <li>Add noodles to mushroom bolognese along with the noodles water. Toss to combine. Divide between 4 bowls and garnish with thai basil. Enjoy!</li>
  </ol>
