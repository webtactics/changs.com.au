ingredients: |
  <h3>Ingredients:</h3>
  <ul>
  <li>2 tbs <a href="/products/Changs-Original-Fried-Noodles/" title="Chang's Original Fried Noodles" alt="Chang's Original Fried Noodles">Chang's Original Fried Noodles</a></li>
  <li>2 small bananas, cut in half down the centre</li>
  <li>1 tbs maple syrup</li>
  <li>1 tsp coconut oil</li>
  <li>&frac12; tsp ground cinnamon</li>
  <li>Vanilla ice cream of choice</li>
  <li>Drizzle of caramel sauce OR maple syrup</li>
  </ul>
method: |
  <h3>Method:</h3>
  <ol>
  <li>Add to a frying pan the maple syrup, coconut oil and cinnamon. Once melted add the bananas open side facing down first and over medium heat fry each side for 1-2 minutes or until golden.</li>
  <li>Prepare to scoop ice cream into bowls and drizzle some caramel sauce or maple syrup over ice cream.</li>
  <li>Place cinnamon bananas over ice cream and then sprinkle with 1 tbs of <a href="/products/Changs-Original-Fried-Noodles/" title="Chang's Original Fried Noodles" alt="Chang's Original Fried Noodles">Chang's Original Fried Noodles</a>. Enjoy</li>
  </ol>
