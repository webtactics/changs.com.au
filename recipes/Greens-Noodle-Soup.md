ingredients: |
  <h3>Ingredients:</h3>
  <ul>
  <li>1 packet <a href="/products/Changs-Long-Life-Noodles/" title="Chang's Long Life Noodles" alt="Chang's Long Life Noodles">Chang's Long Life Noodles</a></li>
  <li>6 cups vegetable stock</li>
  <li>2 tbs Chang&rsquo;s Oyster Sauce</li>
  <li>2 tbs Chang&rsquo;s Soy Sauce</li>
  <li>4 baby bok choy, sliced in half</li>
  <li>8 oyster mushrooms, sliced in half</li>
  <li>8 brown mushrooms, sliced thick</li>
  <li>1 tbs <a href="/products/Changs-Sesame-Oil/" title="Chang's Sesame Oil" alt="Chang's Sesame Oil">Chang's Sesame Oil</a></li>
  <li>Thai Basil</li>
  <li>Green Chilli</li>
  <li>Fresh Lime</li>
  </ul>
method: |
  <h3>Method:</h3>
  <ol>
  <li>Boil noodles and drain. Divide between 4 bowls.</li>
  <li>In a pot add the sesame oil, bok choy and mushrooms. Saut&eacute; for 2 minutes stirring continuously and then add the vegetable stock, oyster sauce and soy sauce. Bring to boil and simmer for 2 minutes.</li>
  <li>Turn off heat and divide evenly the vegetables and broth between the bowls.</li>
  <li>Garnish with thai basil, green chilli slices and a squeeze of lime juice.</li>
  </ol>
