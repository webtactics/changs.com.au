ingredients: |
  <h3>Ingredients:</h3>
  <ul>
  <li>1 x 500ml <a href="/products/Changs-Master-Stock/" title="Chang's Master Stock" alt="Chang's Master Stock">Chang's Master Stock</a></li>
  <li>500ml cold water</li>
  <li>4 x 200g barramundi fillets</li>
  <li>150g yellow and green beans, trimmed</li>
  <li>1 bunch Chinese broccoli, roughly chopped</li>
  <li>1/2 cup loosely packed coriander leaves</li>
  <li>1 x 100g packet <a href="/products/Changs-Crunchy-Fried-Noodles/" title="Chang's Crunchy Fried Noodles" alt="Chang's Crunchy Fried Noodles">Chang's Crunchy Fried Noodles</a></li>
  </ul>
method: |
  <h3>Method:</h3>
  <ol>
  <li>Combine <a href="/products/Changs-Master-Stock/" title="Chang's Master Stock" alt="Chang's Master Stock">Chang's Master Stock</a> and water in a large saucepan and bring to the boil over medium heat.</li>
  <li>Add fish fillets and simmer gently for three minutes.</li>
  <li>Add beans and broccoli and simmer a further two minutes or until fish is cooked through.</li>
  <li>Divide fish fillets, beans and broccoli between four deep serving bowls.</li>
  <li>Ladle over some of the hot stock and garnish with the coriander and crunchy noodles.</li>
  </ol>
