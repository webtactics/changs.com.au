ingredients: |
  <h3>Ingredients:</h3>
  <ul>
  <li>1 x 500ml <a href="/products/Changs-Master-Stock/" title="Chang's Master Stock" alt="Chang's Master Stock">Chang's Master Stock</a></li>
  <li>500ml cold water</li>
  <li>1kg piece pork belly</li>
  <li>150g brown sugar</li>
  <li>1 tbsp. <a href="/products/Changs-Original-Fish-Sauce/" title="Chang's Original Fish Sauce" alt="Chang's Original Fish Sauce">Chang's Original Fish Sauce</a></li>
  <li>1 lime, juiced</li>
  <li>1 large orange, thickly sliced</li>
  <li>1 bunch baby bok choy, thickly sliced, steamed</li>
  <li>1 bunch choy sum, thickly sliced, steamed</li>
  <li>2 tbsp. sesame seeds, toasted</li>
  <li>2 small red chillies, finely sliced</li>
  </ul>
method: |
  <h3>Method:</h3>
  <p>Preheat oven to 160 degrees.</p>
  <ol>
  <li>Combine <a href="/products/Changs-Master-Stock/" title="Chang's Master Stock" alt="Chang's Master Stock">Chang's Master Stock</a> and water in a large saucepan and bring to the boil over high heat.</li>
  <li>Add pork belly, cover and bring back to the boil.</li>
  <li>Place saucepan in the preheated oven for two hours. Remove pork belly from pan and set aside on a tray while you prepare the caramel.</li>
  <li>In a large frying pan heat brown sugar over medium heat and stir until dissolved and caramelised &ndash; around five minutes.</li>
  <li>Add half a cup of the hot stock, fish sauce, lime juice and orange slices.</li>
  <li>Simmer gently for five minutes or until caramel is reduced and thickened.</li>
  <li>Cut pork belly into thick pieces and add to the caramel, tossing gently to coat the pieces.</li>
  <li>Serve with the orange slices, steamed greens, sesame seeds and chilli, and drizzle with remaining caramel.</li>
  </ol>
