ingredients: |
  <h3>Ingredients:</h3>
  <ul>
  <li><span style="line-height: 22.3999996185303px;">2 x 200g blocks white chocolate, chopped</span></li>
  <li><span style="line-height: 22.3999996185303px;">1 x 200g block milk chocolate, chopped</span></li>
  <li><span style="line-height: 22.3999996185303px;">1 packet <a title="Chang's Crunchy Fried Noodles" href="products/Changs-Crunchy-Fried-Noodles/" target="_blank" rel="noopener">Chang's Crunchy Fried Noodles</a></span></li>
  <li><span style="line-height: 22.3999996185303px;">2/3 cup hundreds and thousands</span></li>
  <li><span style="line-height: 22.3999996185303px;">1 cup marshmallows , sliced</span></li>
  </ul>
method: |
  <h3>Method:</h3>
  <ol>
  <li>Line 3 baking trays with non- stick baking paper. Place white and milk chocolates in two heatproof microwave safe bowls. Microwave on medium (50%) for 2-3 minutes or until melted and smooth, stirring with a metal spoon every 30 seconds.</li>
  <li>Spoon 1/4 cup measures of chocolate onto a prepared tray. Repeat three more times to form four rounds.</li>
  <li>Sprinkle generously with noodles, hundreds and thousands and marshmallows. Repeat with remaining chocolate , noodles, hundreds and thousands and marshmallows.</li>
  <li>Set aside 1 hour or until firm.</li>
  </ol>
